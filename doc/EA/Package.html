def initialize(kernel, repo)
  super(kernel, repo)
  @name = @kernel.name
  pid = @kernel.ParentID
  @parent=@repo.get_package(pid) unless pid==0
  begin
    init_collections
  rescue Exception => e
    $stderr.puts "Error initialising collections for '#{kernel.Name}' of type '#{kernel.Type}'"
    $stderr.puts e.message
  end
end
