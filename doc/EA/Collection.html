def add_new(name, type=nil)
  obj=@kernel.AddNew(name,type)
  obj.Update
  @kernel.Refresh
  case @klass.to_s
  when "EA::Element" then
    result=Element.new(obj,@repo,type)
  when "EA::Package" then
    result=Package.new(obj,@repo)
  end
  result
end

Find the index of a given object in this collection

# File lib/eawrapper/collection.rb, line 55
def find(obj)
  match=nil
  n=self.size
  for idx in 0...n do
    if self[idx]==obj then
      match=idx
      break
    end
  end
  match
end

def find_all
  result=[]
  self.each do |e|
    test=if block_given? then yield(e) else true end
    if test then
      result << e
    end
  end
  result
end
