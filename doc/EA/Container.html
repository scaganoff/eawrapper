  def get_element_by_name(name)

#    puts "DEBUG> get_element_by_name('#{name}')"

    a=name.split('/')

    begin; n=a.shift; end while n==''

    if a.empty? then

      e=get_child_element(n)

      return e

    else

      p=get_child_package(n)

      p=get_child_element(n) if p.nil?

      return p.get_element_by_name(a.join('/'))

    end

  end
